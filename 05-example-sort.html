\begin{code} {-@ measure elems @-} elems :: (Ord a) => [a] -> S.Set a elems [] = S.empty elems (x:xs) = addElem x xs {-@ inline addElem @-} addElem :: (Ord a) => a -> [a] -> S.Set a addElem x xs = S.union (S.singleton x) (elems xs) \end{code}
